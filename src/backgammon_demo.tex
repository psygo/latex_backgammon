  \subsection{Black and White}

  \begin{backgammon}[%
    board width                = 15cm,
    board height               = 8cm,
    file height                = 3.5cm,
    file gap                   = 0.15cm,
    side column width          = 1.2cm,
    coordinates gap            = 0.3cm,
    coordinates font size      = 12pt,
    checker size               = 0.75cm,
    checker spacing            = 0.0cm,
    bar spacing                = 0.2cm,
    dark checker border color  = black,
    dark checker border width  = 0.0mm,
    light checker border width = 0.4mm,
    board background color     = white,
    light file color           = gray!24,
    dark file color            = gray!72,
    bar background color       = gray!10,
  ]
    % Doubling Die
    \pic at (\bgLeftColWidth/2, \bgHeight/2) {backgammon doubling die = 2, scale = 0.9, font size = 15pt};

    % Checkers
    \foreach \i in {1, ..., 5} {
      \pic at (point cs:12,\i) {backgammon checker = light};

    }
    \foreach \i in {1, ..., 3} {

    }
    \foreach \i in {1, ..., 3} {

    }
